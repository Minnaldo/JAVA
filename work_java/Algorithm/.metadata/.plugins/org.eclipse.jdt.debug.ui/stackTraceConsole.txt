//두번째 (1번인덱스)가 제일 크다고 가정하고,
		//두번째 (1번인덱스)부터 끝까지 돌면서 더 큰놈이 발견되면 바꾸자.
		max = arr[1];
		max_index = 1;
		for(int i = 1; i < arr.length; i++)
		{
			if (max < arr[i])
			{
				max = arr[i];
				max_index = i;
			}
		}
		
		//1번째와 제일 큰 값 자리를 바꾸자.
		tmp = arr[1];
		arr[1] = arr[max_index];
		arr[max_index] = tmp;
